How it works
Each new joiner moves through four stages from hire date to outcome. The orchestrator runs daily and routes each worker to the right stage based on days remaining.
1
Day 0
Hire
Orchestrator
  • Pulls hire date and probation length from Workday
  • Calculates probation end date
  • Stores worker in tracking table
2
T-30 (30 days before end)
Kick-off
Manager
  • Inbox task: confirm goals, schedule meeting, gather notes
  • If no goals exist, goal-setting task fires automatically
People Partner
  • Inbox task: confirm awareness, flag concerns
Employee
  • Self-assessment form sent via Workday Extend
3
T-7 (7 days before end)
Draft Review
Orchestrator
  • Chases anyone with outstanding tasks
  • Compiles goals + feedback + self-assessment
LLM
  • Drafts the review pack
  • Recommends outcome: Pass / Extend / Escalate
Manager
  • Receives drafted pack as inbox task
4
Day 90
Outcome
Manager
  • Reviews pack, confirms or adjusts the outcome
Orchestrator
  • Writes outcome back to Workday worker record
  • Closes the business process
Daily check: the orchestrator runs every morning and routes each worker into whichever stage their days-remaining count puts them in.

Probation Review Orchestrator

Automated pipeline that tracks every new joiner's probation clock, fires the right tasks to the right people at the right time, and uses an LLM to draft the review pack so managers aren't starting from scratch.

from datetime import date, timedelta

class Worker:
    def __init__(self, worker_id, name, hire_date, probation_days):
        self.worker_id      = worker_id
        self.name           = name
        self.hire_date      = hire_date
        self.probation_days = probation_days
        self.probation_end  = hire_date + timedelta(days=probation_days)

    def days_until_end(self, today=None):
        return (self.probation_end - (today or date.today())).days

# Mock fixtures — 4 workers at different probation stages
def _mock_workers(today):
    return [
        Worker("W001", "Alice Johnson", today - timedelta(days=60), 90),  # T-30
        Worker("W002", "Ben Carter",    today - timedelta(days=83), 90),  # T-7
        Worker("W003", "Carol Smith",   today - timedelta(days=10), 90),  # too early
        Worker("W004", "David Lee",     today - timedelta(days=95), 90),  # ended
    ]

# Live Workers API — paged GET request
def _live_workers():
    client, workers, offset, limit = get_wd_client(), [], 0, 100
    while True:
        data = client.get(f"{WORKDAY_API_BASE}/workers",
                          params={"limit": limit, "offset": offset})
        for w in data.get("data", []):
            workers.append(Worker(
                w["id"],
                w.get("descriptor", "Unknown"),
                date.fromisoformat(w["hireDate"]),
                int(w.get("probationPeriodDays") or 90),
            ))
        if offset + limit >= data.get("total", 0):
            break
        offset += limit
    return workers

def get_workers(today=None):
    today = today or date.today()
    return _mock_workers(today) if USE_MOCK_WORKDAY else _live_workers()

class FeedbackEntry:
    def __init__(self, date_str, from_name, comment, badge=None):
        self.date, self.from_name, self.comment, self.badge = date_str, from_name, comment, badge

    def to_text(self):
        badge = f" [{self.badge}]" if self.badge else ""
        return f"[{self.date}] {self.from_name}{badge}: {self.comment}"

import re, anthropic
from dataclasses import dataclass

@dataclass
class ReviewPackResult:
    review_pack:         str
    recommended_outcome: str
    raw_response:        str

def generate_review_pack(worker_name, probation_end_date, goals, feedback_entries):
    goals_section = (
        "\n".join(f"  - {g.title} | Status: {g.status}"
                  + (f" | Due: {g.due_date}" if g.due_date else "")
                  for g in goals)
        if goals else "  No goals have been set."
    )
    feedback_section = (
        "\n".join(f"  {e.to_text()}" for e in feedback_entries)
        if feedback_entries else "  No feedback entries found."
    )

    prompt = f"""You are an experienced HR business partner preparing a probation review pack.
Your output will be sent directly to the line manager. Write clearly and professionally. Be specific.

EMPLOYEE: {worker_name}
PROBATION END DATE: {probation_end_date}

GOALS SET DURING PROBATION:
{goals_section}

MANAGER AND PEER FEEDBACK (from Workday Feedback):
{feedback_section}

Produce a probation review pack with these exact sections:

## Probation Review Pack — {worker_name}

### 1. Summary
A 2-3 sentence executive summary of this employee's probation performance.

### 2. Goals Assessment
For each goal: state whether it was met, partially met, or not met, and briefly explain why.
If no goals were set, flag this as a significant concern.

### 3. Key Observations
3-5 bullet points drawing on the feedback. Be specific.

### 4. Areas of Concern (if any)
State concerns plainly, or write "No material concerns identified."

### 5. Recommended Outcome
State one of: PASS | EXTEND | ESCALATE
Then give a one-paragraph justification.

End your response with this exact line:
OUTCOME: <PASS|EXTEND|ESCALATE>"""

    client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)
    message = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=2048,
        messages=[{"role": "user", "content": prompt}],
    )
    raw = message.content[0].text

    match   = re.search(r"OUTCOME:\s*(PASS|EXTEND|ESCALATE)", raw, re.IGNORECASE)
    outcome = match.group(1).lower() if match else "unknown"
    pack    = re.sub(r"\nOUTCOME:.*$", "", raw, flags=re.IGNORECASE).strip()

    return ReviewPackResult(review_pack=pack, recommended_outcome=outcome, raw_response=raw)
